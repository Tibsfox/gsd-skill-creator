% ---------- Callout boxes ----------
\tcbset{
  gsdcalloutbase/.style={
    enhanced,
    breakable,
    left=10pt,
    right=8pt,
    top=6pt,
    bottom=6pt,
    boxrule=0pt,
    frame hidden,
    borderline west={4pt}{0pt}{#1},
    colback=#1,
    fonttitle=\bfseries\headingfont,
    fontupper=\small,
    arc=2pt
  }
}
