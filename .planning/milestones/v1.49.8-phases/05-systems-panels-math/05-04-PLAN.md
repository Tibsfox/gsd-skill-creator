```typescript
export interface RosettaConcept {
  id: string;
  name: string;
  domain: string;
  description: string;
  panels: Map<PanelId, PanelExpression>;
  relationships: ConceptRelationship[];
  calibration?: CalibrationProfile;
  complexPlanePosition?: ComplexPosition;
}

export interface ComplexPosition {
  real: number;       // concreteness axis (-1 to 1)
  imaginary: number;  // complexity axis (-1 to 1)
  magnitude: number;  // Math.sqrt(real^2 + imaginary^2)
  angle: number;      // Math.atan2(imaginary, real) in radians
}

export interface ConceptRelationship {
  type: 'dependency' | 'analogy' | 'cross-reference';
  targetId: string;
  description: string;
}

export type PanelId = 'python' | 'cpp' | 'java' | 'lisp' | 'pascal' | 'fortran' | 'perl' | 'algol' | 'unison' | ...;
```
