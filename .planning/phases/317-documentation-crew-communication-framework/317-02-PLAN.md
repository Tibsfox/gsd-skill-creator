This is the master priority arbitration configuration. It defines how messages from different loops are ordered when multiple messages compete for processing.

```yaml
name: bus-arbitration
version: 1.0.0
description: "Priority-based message arbitration for the mission communication bus. When multiple messages arrive simultaneously, higher-priority messages are processed first."

# Priority scheme -- lower number = higher priority
# Matches LoopPriority type in src/types/openstack.ts (0-7)
priority_levels:
  0:
    name: HALT
    description: "Emergency stop -- immediate processing, preempts everything"
    behavior: "Interrupt current processing, deliver to all agents immediately"
    source: "FLIGHT only"
  1:
    name: COMMAND
    description: "Mission-level directives from FLIGHT"
    behavior: "Process after any active HALT, before all other messages"
    loops: [command]
  2:
    name: CRITICAL_OPERATIONAL
    description: "Core operational messages that affect mission progress"
    behavior: "Round-robin among same-priority loops"
    loops: [execution, user, health, cloud-ops]
  3:
    name: COORDINATION
    description: "Coordination and resource management messages"
    behavior: "Process after all priority 2 messages"
    loops: [specialist, budget]
  4:
    name: MAINTENANCE
    description: "System maintenance and documentation upkeep"
    behavior: "Process when no higher-priority messages pending"
    loops: [doc-sync]
  5:
    name: ROUTINE
    description: "Reserved for future routine loops"
    behavior: "Background processing"
    loops: []
  6:
    name: BACKGROUND
    description: "Non-critical background tasks"
    behavior: "Process only when bus is idle"
    loops: [observation]
  7:
    name: HEARTBEAT
    description: "Keep-alive and status monitoring"
    behavior: "Lowest priority, dropped under load"
    loops: []

# Arbitration rules
rules:
  - name: "halt-preemption"
    description: "Priority 0 (HALT) preempts all in-progress message processing"
    condition: "message.priority == 0"
    action: "interrupt-and-deliver"

  - name: "same-priority-ordering"
    description: "Messages at the same priority level are processed round-robin by loop"
    condition: "message.priority == current.priority"
    action: "round-robin"

  - name: "starvation-prevention"
    description: "No loop may be starved for more than 5 cycles"
    condition: "loop.cycles_without_processing > 5"
    action: "promote-priority-by-1"

  - name: "budget-block-enforcement"
    description: "Budget block messages (priority 3) are promoted to priority 1 when block threshold reached"
    condition: "message.type == 'budget-block'"
    action: "promote-to-priority-1"

# Bus capacity
capacity:
  max_in_flight_messages: 100
  max_message_size_bytes: 65536
  queue_overflow_action: "drop-lowest-priority"
```

```yaml
name: halt-signal
version: 1.0.0
description: "HALT signal configuration -- priority 0 emergency stop that propagates to all agents within 1 communication cycle. No partial operations allowed."

# HALT is not a loop -- it is a cross-cutting signal that uses priority 0
# to preempt all other bus traffic
signal:
  priority: 0
  type: "broadcast-immediate"
  authorized_senders:
    - role: FLIGHT
      reason: "Mission director can halt all operations"
    - role: BUDGET
      condition: "token_consumption >= 95%"
      reason: "Budget exhaustion triggers automatic HALT to prevent uncontrolled spending"
    - role: SURGEON
      condition: "emergency_health_event"
      reason: "Critical system health failure triggers automatic HALT"

propagation:
  method: "interrupt-all"
  max_cycles: 1
  delivery_guarantee: "all-or-none"
  description: "HALT signal is delivered to every active agent in the current crew within a single communication cycle. If any agent cannot be reached, the entire crew is considered halted (fail-safe)."

  target_agents: "all active agents in current crew"
  acknowledgment_required: true
  acknowledgment_timeout_seconds: 5

# What agents must do when they receive HALT
agent_behavior:
  immediate_actions:
    - "Stop accepting new task assignments"
    - "Complete current atomic operation if safe (< 5 seconds remaining)"
    - "Abort current operation if not safe to complete"
    - "Send acknowledgment to FLIGHT"
  prohibited_actions:
    - "Starting new tasks"
    - "Sending non-HALT messages on any loop"
    - "Modifying system state"
    - "Making API calls to OpenStack services (except read-only health checks)"

  no_partial_operations:
    description: "HALT ensures no operation is left in a partial state"
    mechanism: "Each EXEC task is designed as an atomic unit. If HALT arrives mid-task, EXEC completes the current atomic step then stops. Rollback procedures are available for each atomic step."
    examples:
      - "Kolla-Ansible deploy step: complete current container, do not start next"
      - "Documentation write: save current file, do not start next section"
      - "Verification test: complete current test, report partial results"

# HALT recovery
recovery:
  resume_authority: FLIGHT
  resume_signal: "RESUME"
  resume_priority: 1
  pre_resume_checks:
    - "SURGEON confirms system health"
    - "BUDGET confirms resource availability"
    - "All agents acknowledge RESUME readiness"
  post_resume_action: "FLIGHT re-broadcasts current phase status to all agents"

# Integration with budget block

  description: "When BUDGET detects 95% consumption, it sends HALT on behalf of resource conservation. This is an automatic HALT -- no FLIGHT approval needed."
  recovery_condition: "FLIGHT approves continued execution with reduced scope or increased budget"
```
