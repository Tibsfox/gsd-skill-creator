Python is dynamically typed, garbage-collected, multi-paradigm, and designed around one central principle: readability counts. The significant indentation that confuses newcomers is a deliberate rejection of brace-based syntax — a commitment to making the visual structure of the code match its logical structure. You cannot write obfuscated Python the way you can write obfuscated Perl or C. The language won’t let you.

Everything Is An Object

In Python, everything is an object. Integers, strings, functions, classes, modules, types — all are first-class objects with attributes, methods, and identity. This uniformity is the foundation of Python’s data model.

# Functions are objects
def greet(name):
    return f"Hello, {name}!"

greet.author = "Guido"           # attach attribute to function
print(greet.__name__)            # "greet"
print(type(greet))               # <class 'function'>

# Classes are objects
class Dog:
    pass

Dog.species = "Canis lupus"      # attach attribute to class
print(type(Dog))                 # <class 'type'>
print(type(type))                # <class 'type'> (type is its own type)

Comprehensions — Python’s Signature Syntax

# List comprehension
squares = [x**2 for x in range(10)]

# With filter
evens_squared = [x**2 for x in range(20) if x % 2 == 0]

# Dict comprehension
lengths = {word: len(word) for word in ["hello", "world", "python"]}

# Set comprehension
unique_lengths = {len(word) for word in words}

# Generator expression (lazy — no list built)
total = sum(x**2 for x in range(1_000_000))

# Nested comprehension
matrix = [[i*j for j in range(5)] for i in range(5)]

# Walrus operator (3.8+) — assign within expression
filtered = [y for x in data if (y := process(x)) is not None]

Comprehensions are Python’s answer to map/filter from functional languages, but with better readability. The Zen says “flat is better than nested” — comprehensions let you flatten what would otherwise be loop-and-append patterns into single expressions.

Classes & Object-Oriented Programming

class Point:
    """A 2D point."""

    # Class variable (shared)
    dimension = 2

    def __init__(self, x: float, y: float):
        self.x = x
        self.y = y

    def distance_to(self, other: 'Point') -> float:
        return ((self.x - other.x)**2 + (self.y - other.y)**2) ** 0.5

    def __repr__(self):
        return f"Point({self.x}, {self.y})"

    def __eq__(self, other):
        return isinstance(other, Point) and (self.x, self.y) == (other.x, other.y)

    def __hash__(self):
        return hash((self.x, self.y))

Dataclasses (PEP 557, Python 3.7+)

from dataclasses import dataclass

@dataclass(frozen=True)
class Point:
    x: float
    y: float

    def distance_to(self, other: 'Point') -> float:
        return ((self.x - other.x)**2 + (self.y - other.y)**2) ** 0.5

# __init__, __repr__, __eq__, __hash__ auto-generated from type hints
p = Point(3, 4)
print(p)                          # Point(x=3, y=4)

Iterators & Generators

# Generator function (uses yield)
def fibonacci():
    a, b = 0, 1
    while True:
        yield a
        a, b = b, a + b

# Take first 10 Fibonacci numbers
from itertools import islice
first_ten = list(islice(fibonacci(), 10))

# Generator expression (lazy)
squares_gen = (x**2 for x in range(1_000_000))  # no memory allocated
total = sum(squares_gen)                         # compute lazily

# itertools — the functional toolbox
from itertools import chain, cycle, groupby, product, combinations
for a, b in combinations(items, 2):             # all pairs
    compare(a, b)

Decorators — Functions That Transform Functions

import time
from functools import wraps

def timed(func):
    @wraps(func)
    def wrapper(*args, **kwargs):
        start = time.perf_counter()
        result = func(*args, **kwargs)
        print(f"{func.__name__} took {time.perf_counter() - start:.3f}s")
        return result
    return wrapper

@timed
def slow_operation(n):
    return sum(i*i for i in range(n))

# Decorator with arguments
def retry(attempts=3):
    def decorator(func):
        @wraps(func)
        def wrapper(*args, **kwargs):
            for i in range(attempts):
                try:
                    return func(*args, **kwargs)
                except Exception as e:
                    if i == attempts - 1:
                        raise
        return wrapper
    return decorator

@retry(attempts=5)
def fetch_url(url):
    ...

Context Managers — Resource Management

# The with statement guarantees cleanup
with open("file.txt", "r") as f:
    content = f.read()
# file is closed here, even if an exception occurred

# Multiple context managers
with open("input.txt") as fin, open("output.txt", "w") as fout:
    fout.write(process(fin.read()))

# Custom context manager with contextlib
from contextlib import contextmanager

@contextmanager
def database_transaction(db):
    tx = db.begin()
    try:
        yield tx
        tx.commit()
    except Exception:
        tx.rollback()
        raise

async / await (Python 3.5+)

import asyncio
import aiohttp

async def fetch(session, url):
    async with session.get(url) as response:
        return await response.text()

async def fetch_all(urls):
    async with aiohttp.ClientSession() as session:
        tasks = [fetch(session, url) for url in urls]
        return await asyncio.gather(*tasks)

# Run concurrently — all requests in flight at once
results = asyncio.run(fetch_all(["https://example.com", ...]))

Python’s async/await (PEP 492, Python 3.5, 2015) was a long time coming. Earlier solutions included Twisted (2002), Tornado (2009), and yield from-based coroutines in 3.4. Native async made Python competitive with Node.js for I/O-bound workloads. FastAPI (2018) is the framework most associated with modern async Python.

Pattern Matching (Python 3.10+)

# Structural pattern matching — not just a switch statement
def handle(event):
    match event:
        case {"type": "click", "x": x, "y": y}:
            return f"clicked at ({x}, {y})"
        case {"type": "key", "key": key} if key.isalpha():
            return f"typed letter {key}"
        case Point(x=0, y=0):
            return "origin"
        case Point(x=x, y=y):
            return f"point at ({x}, {y})"
        case [first, *rest]:
            return f"list with {len(rest)+1} items"
        case _:
            return "unknown"

Type Hints (PEP 484, 2014)

from typing import Optional, Union, Callable, TypeVar, Generic

T = TypeVar('T')

def find(items: list[T], predicate: Callable[[T], bool]) -> Optional[T]:
    for item in items:
        if predicate(item):
            return item
    return None

class Stack(Generic[T]):
    def __init__(self):
        self._items: list[T] = []
    def push(self, item: T) -> None:
        self._items.append(item)
    def pop(self) -> T:
        return self._items.pop()

# Use of PEP 604 union syntax (3.10+)
def parse(value: str | int | None) -> int:
    ...

Type hints are optional and not enforced at runtime. They exist for static analysis tools (mypy, pyright, pyre) and IDEs. The Python community debated type hints fiercely when they were introduced — the gradual, optional approach (unlike Java or Rust) preserved Python’s dynamic nature while allowing teams who wanted static checking to have it.
