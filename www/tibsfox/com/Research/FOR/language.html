! Modern Fortran
Fortran 90 onward shares almost nothing with the FORTRAN of the punched-card era. Free-form, modular, object-capable, strongly typed, with native array semantics, coarrays, and direct C interoperability.

1. Source Form

Fortran has two source forms. Fixed form (FORTRAN 77 and earlier, still legal) is column-sensitive. Free form (Fortran 90+) is what all modern code uses.

! Free-form modern Fortran — no column rules
program hello
  implicit none
  integer :: i
  integer, parameter :: n = 10
  do i = 1, n
    write(*, '(A, I3, A, I5)') ' i=', i, '  i**2=', i*i
  end do
end program hello

Compilers choose source form by extension: .f/.for = fixed, .f90/.f03/.f08/.f18 = free. Fortran is case insensitive — FOO, foo, and Foo are identical. Modern convention: all lowercase.

Line continuation: & at end of line. Comments: ! anywhere on the line.

2. Program Structure

UnitPurpose
programMain entry point. Exactly one per executable.
moduleReusable collection of data, types, procedures. The modern unit of encapsulation.
subroutineProcedure returning nothing (like void).
functionProcedure returning a value.
submoduleSplit module implementation across files (Fortran 2008+).
 
module math_utils
  implicit none
  private                              ! default: everything private
  public :: square, cube              ! explicit public API

contains

  pure function square(x) result(y)
    real, intent(in) :: x
    real :: y
    y = x * x
  end function square

  pure function cube(x) result(y)
    real, intent(in) :: x
    real :: y
    y = x * x * x
  end function cube

end module math_utils

program use_math
  use math_utils, only: square, cube   ! selective import
  implicit none
  print *, square(3.0), cube(3.0)      ! 9.0  27.0
end program use_math

IMPLICIT NONE: Historically, Fortran had a rule that variables starting with I-N were implicit INTEGER, others REAL. Modern code ALWAYS starts with implicit none to disable this and require explicit declarations. Use -fimplicit-none to enforce globally.

3. Data Types & KIND Parameters

The five intrinsic types: integer, real, complex, logical, character. Each has a KIND parameter controlling precision:

use, intrinsic :: iso_fortran_env, only: int32, int64, real32, real64, real128
implicit none

integer(int32)  :: i32              ! 32-bit integer
integer(int64)  :: i64              ! 64-bit integer
real(real32)    :: r32              ! single precision
real(real64)    :: r64              ! double precision
real(real128)   :: r128             ! quad precision (if supported)

complex(real64) :: z                ! double-precision complex
logical         :: flag
character(len=20) :: name

! Selecting kind by required range:
integer, parameter :: ik = selected_int_kind(18)   ! ~18 digits
integer, parameter :: rk = selected_real_kind(15)  ! ~15 digits precision

4. Derived Types

Fortran’s equivalent of C structs — with type-bound procedures, inheritance, and polymorphism.

module point_module
  implicit none

  type :: point
    real :: x, y
  contains
    procedure :: distance_to     ! type-bound method
    procedure :: magnitude
  end type point

contains

  pure function distance_to(self, other) result(d)
    class(point), intent(in) :: self, other
    real :: d
    d = sqrt((self%x - other%x)**2 + (self%y - other%y)**2)
  end function distance_to

  pure function magnitude(self) result(m)
    class(point), intent(in) :: self
    real :: m
    m = sqrt(self%x**2 + self%y**2)
  end function magnitude

end module point_module

program use_points
  use point_module
  implicit none
  type(point) :: p1, p2
  p1 = point(1.0, 2.0)
  p2 = point(4.0, 6.0)
  print *, 'Distance:', p1%distance_to(p2)       ! 5.0
  print *, 'Magnitude:', p1%magnitude()           ! 2.236
end program use_points

6. Object-Oriented Fortran (2003+)

module shapes
  implicit none

  ! Abstract base type
  type, abstract :: shape
  contains
    procedure(area_interface), deferred :: area
  end type shape

  abstract interface
    pure function area_interface(self) result(a)
      import :: shape
      class(shape), intent(in) :: self
      real :: a
    end function area_interface
  end interface

  ! Concrete: circle extends shape
  type, extends(shape) :: circle
    real :: radius
  contains
    procedure :: area => circle_area
  end type circle

  ! Concrete: rectangle extends shape
  type, extends(shape) :: rectangle
    real :: width, height
  contains
    procedure :: area => rectangle_area
  end type rectangle

contains

  pure function circle_area(self) result(a)
    class(circle), intent(in) :: self
    real :: a
    a = 3.14159265 * self%radius**2
  end function circle_area

  pure function rectangle_area(self) result(a)
    class(rectangle), intent(in) :: self
    real :: a
    a = self%width * self%height
  end function rectangle_area

end module shapes

program polymorphism_demo
  use shapes
  implicit none
  class(shape), allocatable :: s

  s = circle(radius=5.0)
  print *, 'Circle area:', s%area()        ! polymorphic dispatch

  deallocate(s)
  s = rectangle(width=4.0, height=3.0)
  print *, 'Rectangle area:', s%area()
end program polymorphism_demo

7. Control Structures

! IF / THEN / ELSE
if (x > 0.0) then
  print *, 'positive'
else if (x < 0.0) then
  print *, 'negative'
else
  print *, 'zero'
end if

! DO loops
do i = 1, 10;      print *, i;  end do
do i = 1, 100, 2;  print *, i;  end do      ! step 2
do i = 100, 1, -1; print *, i;  end do      ! countdown

! DO WHILE
do while (residual > 1.0e-6)
  call iterate(x, residual)
end do

! Named loops with exit / cycle
outer: do i = 1, n
  inner: do j = 1, m
    if (bad(i, j)) exit outer
    if (skip(i, j)) cycle inner
    call process(i, j)
  end do inner
end do outer

! SELECT CASE
select case (status)
case (1)
  print *, 'one'
case (2:5)
  print *, 'two to five'
case (10, 20, 30)
  print *, 'ten twenty or thirty'
case default
  print *, 'other'
end select

8. C Interoperability (ISO_C_BINDING)

Fortran 2003 introduced the iso_c_binding intrinsic module, providing standard, portable C interoperability.

module c_interop_example
  use, intrinsic :: iso_c_binding
  implicit none

  ! Declare a C function callable from Fortran
  interface
    function c_sqrt(x) bind(C, name="sqrt")
      import :: c_double
      real(c_double), value :: x
      real(c_double) :: c_sqrt
    end function c_sqrt
  end interface

contains

  ! A Fortran function callable from C
  function fortran_square(x) bind(C, name="fortran_square") result(y)
    real(c_double), value :: x
    real(c_double) :: y
    y = x * x
  end function fortran_square

end module c_interop_example

9. I/O

! Formatted output
print *, 'Simple unformatted'
print '(A, I5, F10.4)', 'Value:', 42, 3.14159
write(*, '(A, I0)') 'i = ', i

! File I/O
integer :: unit, ios
open(newunit=unit, file='data.txt', status='old', action='read', iostat=ios)
if (ios /= 0) stop 'Cannot open file'

read(unit, *, iostat=ios) x, y, z
close(unit)

! Namelist I/O — easy structured input
integer :: n
real    :: tolerance
character(len=80) :: method
namelist /config/ n, tolerance, method

open(newunit=unit, file='config.nml', action='read')
read(unit, nml=config)
close(unit)

  • Better C interop — more C types, cleaner binding
